import pandas as pd
import matplotlib.pyplot as plt

from io import StringIO
import requests

# Descargar datos del ONI de NOAA
url = "https://www.cpc.ncep.noaa.gov/data/indices/oni.ascii.txt"

# Procesar los datos

# Crear una columna de fecha combinando a√±o y estaci√≥n

    'DJF': 1, 'JFM': 2, 'FMA': 3, 'MAM': 4, 'AMJ': 5, 'MJJ': 6,
    'JJA': 7, 'JAS': 8, 'ASO': 9, 'SON': 10, 'OND': 11, 'NDJ': 12
}

# Filtrar √∫ltimos 40 a√±os

# Crear la gr√°fica

# Colorear las √°reas seg√∫n El Ni√±o (positivo) y La Ni√±a (negativo)

                where=(df_filtered['ANOM'] > 0), 
                color='red', alpha=0.5, label='El Ni√±o (c√°lido)')
ax.fill_between(df_filtered['Date'], df_filtered['ANOM'], 0, 
                where=(df_filtered['ANOM'] < 0), 
                color='blue', alpha=0.5, label='La Ni√±a (fr√≠o)')

# L√≠nea de tendencia

# L√≠neas de referencia para umbrales de El Ni√±o/La Ni√±a
ax.axhline(y=0.5, color='red', linestyle='--', linewidth=0.8, alpha=0.5, label='Umbral El Ni√±o (+0.5¬∞C)')
ax.axhline(y=-0.5, color='blue', linestyle='--', linewidth=0.8, alpha=0.5, label='Umbral La Ni√±a (-0.5¬∞C)')
ax.axhline(y=0, color='gray', linestyle='-', linewidth=0.5)

ax.set_title('√çndice Oce√°nico Ni√±o (ONI) - √öltimos 40 a√±os\nFuente: NOAA Climate Prediction Center', 
             fontsize=14, fontweight='bold')

plt.show()

# Mostrar estad√≠sticas
print(f"\nüìä Estad√≠sticas del ONI ({start_year}-{current_year}):")

print(f"Anomal√≠a m√≠nima (La Ni√±a m√°s fuerte): {df_filtered['ANOM'].min():.2f}¬∞C en {df_filtered[df_filtered['ANOM'] == df_filtered['ANOM'].min()]['Date'].dt.strftime('%Y-%m').values[0]}")
